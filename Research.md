- class Kitablar:]
  - def __init_(kitab,_adi,_ili,_yazicisi,_qiymeti):
    - kitab.ad=_adi
    - kitab.ili=_ili
    - kitab.yazicisi=_yazicisi
    - kitab.qiymeti=_qiymeti
    - books.append(kitab)
  - def Informasiyayiqma(kitab):
